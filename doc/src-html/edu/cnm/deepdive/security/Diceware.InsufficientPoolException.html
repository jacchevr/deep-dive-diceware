001package edu.cnm.deepdive.security;
002
003import java.io.BufferedReader;
004import java.io.File;
005import java.io.FileInputStream;
006import java.io.FileNotFoundException;
007import java.io.IOException;
008import java.io.InputStreamReader;
009import java.security.NoSuchAlgorithmException;
010import java.security.SecureRandom;
011import java.util.ArrayList;
012import java.util.Collection;
013import java.util.Enumeration;
014import java.util.LinkedList;
015import java.util.List;
016import java.util.Random;
017import java.util.ResourceBundle;
018import java.util.regex.Matcher;
019import java.util.regex.Pattern;
020
021/**
022 * The <code>Diceware</code> class implements a Diceware-based passphrase generator, using a word
023 * list provided in the constructor invocation. If a pseudo-random number generator is not set
024 * (using the {@link Diceware#setRng(Random)} method), then an instance of {@link SecureRandom} is
025 * created and used for selecting words at random from the list.
026 * 
027 * @author Nicholas Bennett with Deep Dive Coding Java Cohort 3
028 * @version 0.9
029 */
030public class Diceware {
031
032  private static final String DEFAULT_RESOURCE_BUNDLE = "wordlist";
033  private static final String NEGATIVE_PASSPHRASE_MESSAGE = "Passphrase length must be positive.";
034  private static final String LINE_PATTERN = "^\\s*(\\d+)\\s+(\\S+)\\s*$";
035
036  private List<String> words;
037  private Random rng = null;
038
039  /**
040   * Initializes an instance of <code>Diceware</code> using a reference to a {@link java.io.File}
041   * object. If the <code>File</code> does not exist, of cannot be read, an exception will be
042   * thrown.
043   * 
044   * @param file file to read for word list.
045   * @throws FileNotFoundException if file does not exist.
046   * @throws IOException if file can't be read.
047   */
048
049  public Diceware() {
050    this(ResourceBundle.getBundle(DEFAULT_RESOURCE_BUNDLE));
051  }
052
053  public Diceware(File file) throws FileNotFoundException, IOException {
054    words = new ArrayList<>();
055    try (FileInputStream input = new FileInputStream(file);
056        InputStreamReader reader = new InputStreamReader(input);
057        BufferedReader buffer = new BufferedReader(reader);) {
058      Pattern p = Pattern.compile(LINE_PATTERN);
059      for (String line = buffer.readLine(); line != null; line = buffer.readLine()) {
060        Matcher m = p.matcher(line);
061        if (m.matches()) {
062          words.add(m.group(2));
063        }
064      }
065    }
066  }
067
068  /**
069   * Initializes an instance of <code>Diceware</code> using a {@link Collection} as the source of
070   * words for the word list.
071   * 
072   * @param source word list source.
073   */
074  public Diceware(Collection<String> source) {
075    words = new ArrayList<>(source);
076  }
077
078  /**
079   * Initializes an instance of <code>Diceware</code> using a {@link ResourceBundle} object as the
080   * source of words for the word list. (The property values from the <code>ResourceBundle</code>
081   * are taken as the words; the property names/keys are ignored.)
082   * 
083   * @param bundle properties provided words (values) for word list.
084   */
085  public Diceware(ResourceBundle bundle) {
086    words = new ArrayList<>();
087    Enumeration<String> en = bundle.getKeys();
088    while (en.hasMoreElements()) {
089      words.add(bundle.getString(en.nextElement()));
090    }
091  }
092
093  /**
094   * Initializes (if necessary) and returns the {@link Random} instance to be used for selecting
095   * words from the word list.
096   * 
097   * @return pseudo-random number generator instance.
098   * @throws NoSuchAlgorithmException if lazy initialization is used, and default strong provider
099   *         algorithm does not exist.
100   */
101  public Random getRng() throws NoSuchAlgorithmException {
102    if (rng == null) {
103      rng = SecureRandom.getInstanceStrong();
104    }
105    return rng;
106  }
107
108  /**
109   * Sets a reference to the {@link Random} instance to be used for selecting words from the word
110   * list.
111   * 
112   * @param rng pseudo-random number generator instance.
113   */
114  public void setRng(Random rng) {
115    this.rng = rng;
116  }
117
118  /**
119   * Generates and returns (in a <code>String[]</code>) a password of the specified length. The
120   * inclusion of duplicates is controlled by the <code>duplicatesAllowed</code> argument. If the
121   * specified length is greater than the number of words in the word list, and duplicates aren't
122   * allowed, then an infinite loop will result.
123   * 
124   * @param length number of words to include in generated passphrase.
125   * @param duplicatesAllowed true if duplicate words are allowed; false otherwise.
126   * @return words in generated passphrase.
127   * @throws NoSuchAlgorithmException if algorithm for default strong source of randomness is not
128   *         available.
129   * @throws InsufficientPoolException if password length exceeds word list, and duplicates not
130   *         allowed or word list has no words.
131   * @throws IllegalArgumentException if requested length is not positive.
132   */
133  public String[] generate(int length, boolean duplicatesAllowed)
134      throws NoSuchAlgorithmException, InsufficientPoolException, IllegalArgumentException {
135    if (length <= 0) {
136      throw new IllegalArgumentException(NEGATIVE_PASSPHRASE_MESSAGE);
137    }
138    if ((words.size() == 0 && length > 0) || (!duplicatesAllowed && length > words.size())) {
139      throw new InsufficientPoolException();
140    }
141    List<String> passphrase = new LinkedList<>();
142    while (passphrase.size() < length) {
143      String word = generate();
144      if (duplicatesAllowed || !passphrase.contains(word)) {
145        passphrase.add(word);
146      }
147    }
148    return passphrase.toArray(new String[passphrase.size()]);
149  }
150
151  /**
152   * Generates and returns (in a <code>String[]</code>) a password of the specified length. This
153   * method simply invokes {@link #generate(int, boolean) generate(length, true)} &ndash; that is,
154   * it invokes {@link #generate(int, boolean)}, specifying that duplicates are allowed.
155   * 
156   * @param length number of words to include in generated passphrase.
157   * @return words in generated passphrase.
158   * @throws NoSuchAlgorithmException if algorithm for default strong source of randomness is not
159   *         available.
160   * @throws InsufficientPoolException if word list has no words.
161   * @throws IllegalArgumentException if requested length is negative.
162   */
163  public String[] generate(int length)
164      throws NoSuchAlgorithmException, InsufficientPoolException, IllegalArgumentException {
165    return generate(length, true);
166  }
167
168  public String generate(int length, String delimiter)
169      throws InsufficientPoolException, NoSuchAlgorithmException, IllegalArgumentException {
170    return generate(length, delimiter, true);
171  }
172
173  public String generate(int length, String delimiter, boolean duplicatesAllowed)
174      throws InsufficientPoolException, NoSuchAlgorithmException, IllegalArgumentException {
175    String[] words = generate(length, duplicatesAllowed);
176    StringBuilder builder = new StringBuilder(words[0]);
177    for (int i = 1; i < words.length; i++) {
178      builder.append(delimiter);
179      builder.append(words[i]);
180    }
181    return builder.toString();
182  }
183
184  private String generate() throws NoSuchAlgorithmException {
185    int index = getRng().nextInt(words.size());
186    return words.get(index);
187  }
188
189  public static class InsufficientPoolException extends IllegalArgumentException {
190
191    private InsufficientPoolException() {
192
193    }
194
195  }
196
197}
